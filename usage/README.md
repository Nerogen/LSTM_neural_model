## _Example of usage_
### Configuration of config file
    length_of_sequence = 50
    start_value = 0
    end_value = 100
    steps_on_range = 1000
    number_of_epochs = 10
    batch_size_of_sequence = 10
    train_size_of_sequence = 0.8
    activation_function = "relu"
